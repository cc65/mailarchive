int OpenOutput(char *fn,int fnr){
char cmd[64];
int rc;
 sprintf(cmd,"s:%s",fn);
 rc=cbm_open(15,A_DVN,15,cmd);
 cbm_close(15);
 sprintf(cmd,"%s,s,w",fn);
 rc=cbm_open(fnr,A_DVN,fnr,cmd);
 if(rc){
  DispError("Open-Out-Fehler");
  return 1;
 }
 return 0;
}
