Hi Matt,
 
> Would there be any way to create a zero page pointer?

This worked for me:

#include <stdio.h>

const char msg[] = "Hello.";

#pragma bssseg ("ZEROPAGE")
static char *pMsg;
#pragma bssseg ("DATA")
static char *pCheck;

main(int c,char**v) {
pMsg = (char *)msg;
pCheck = pMsg;
printf("%s %s (%c)\n",msg,pMsg,*pCheck);
getchar();
}
