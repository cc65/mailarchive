A macro will do what you want to do:

#define MAKE_WORD(high,low) (((high) << 8) | (low))

extern void fastcall setcharcolors(unsigned int c);

int main(void) {

  setcharcolors(MAKE_WORD(1,0));
  return 0;
  }
