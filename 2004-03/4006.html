Is there a way to force the complier to place the CLD before the jsr decsp1?


0011E6r 1



           ; ---------------------------------------------------------------
0011E6r 1               ; void __near__ Int0 (void)
0011E6r 1



           ; ---------------------------------------------------------------
0011E6r 1
0011E6r 1               .segment "CODE"
0011E6r 1
0011E6r 1               .proc _Int0: near
0011E6r 1
0011E6r 1               .segment "CODE"
0011E6r 1
0011E6r 1               ;
0011E6r 1               ; asm("cld");
0011E6r 1               ;
0011E6r 1  20 rr rr      jsr     decsp1
0011E9r 1  D8            cld
0011EAr 1               ;
