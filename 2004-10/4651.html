Here's the Makefile I use for new cc65 projects:

----------------------------------------------------------------------

TARGET=c64
CC=cl65
AS=ca65
LD=cl65
CFLAGS=-Oirs -t $(TARGET)
AFLAGS=
LDFLAGS=-t $(TARGET)

%.o: %.c
	$(CC) -c $(CFLAGS) $<

%.o: %.s
	$(AS) $(AFLAGS) $<


all: myprogram.prg


OBJS = \
	main.o \
	gfx.o \
	map.o


myprogram.prg: $(OBJS)
	$(LD) $(LDFLAGS) -m myprogram.map -o myprogram.prg $^


clean:
	rm -f *.o
	rm -f myprogram.prg myprogram.map


distclean: clean
	rm -f *~

----------------------------------------------------------------------

Just change the program name and add your files to OBJS.
