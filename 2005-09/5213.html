>>>>> "RW" == Raj Wurttemberg <rajw@c64.us> writes:

RW> Does anyone have an example of reading the 15x1 error channel on a
RW> C64 with cc65?

#include <stdio.h>
#include <cbm.h>

int main(void) {
  unsigned char buf[128];
  unsigned char msg[100]; // should be large enough for all messages
  unsigned char e, t, s;

  if (cbm_open(1, 8, 15, "") == 0) {
    if (cbm_read(1, buf, sizeof(buf)) < 0) {
      puts("read error");
      return(1);
    }
    cbm_close(1);
  }
  if (sscanf(buf, "%hhu, %[^,], %hhu, %hhu", &e, msg, &t, &s) != 4) {
    puts("parse error");
    puts(buf);
    return(2);
  }
  printf("%hhu,%s,%hhu,%hhu\n", (int) e, msg, (int) t, (int) s);
  return(0);
}
