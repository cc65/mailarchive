I noticed that cc65 treated externs inside functions
like statics, meaning that storage was allocated and
initialization was wrongly permitted.

Fix:

In locals.c, ParseOneDecl,

       	} else if ((SC & (SC_STATIC | SC_EXTERN)) ==
SC_STATIC) {
            /* Static variable */
            SymData = ParseStaticDecl (&Decl, &SC);
       	} else if ((SC & (SC_STATIC | SC_EXTERN)) ==
(SC_STATIC | SC_EXTERN)) {
            /* Extern variable */
		    if (CurTok.Tok == TOK_ASSIGN) {
				Error ("Can't initialize extern");
			}
