Another (very similar) problem example...

extern char foo[4];
void setfoo(char a)
{
    foo[2] = a;
}

2.19.9:

.proc   _setfoo: near
    lda     #<(_foo)
    ldx     #>(_foo)
    jsr     pushax
    ldy     #$02
    ldx     #$00
    lda     (sp),y
    ldy     #$02
    jsr     staspidx
    jsr     incsp1
    rts

v2.19.0:
.proc   _setfoo: near
    ldy     #$00
    ldx     #$00
    lda     (sp),y
    sta     _foo+2
    jsr     incsp1
    rts

...and couldn't the last jsr/rts pair be changed to a jmp?
