Something very simple, just adding up longs:

--------------------------<PUT CHAINSAW HERE>---------------------
#include <stdio.h>
#include <time.h>

void main(void)
{
	unsigned long a,b;
	unsigned int i,j;
	clock_t starttime,endtime;

	starttime = clock();
	// Bench starts here
	for(i=0;i<500;++i)
	{
		//printf("i:%d\n",i);
		a = 2;
		b = 1;
		for(j=0;j<20;++j)
		{
			b = b + a;
			a = b;
			//printf("j:%d a:%ld\n",j,a);
		}
	}
	// Bench ends here
	endtime = clock();

	printf("Time in system ticks:%ld\n",endtime-starttime);
	printf("Time in seconds:%ld\n",(endtime-starttime)/CLOCKS_PER_SEC);
}
--------------------------<PUT CHAINSAW HERE>---------------------

Hope it helps :)
