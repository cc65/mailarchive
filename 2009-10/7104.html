#include <assert.h>
#include <string.h>

typedef unsigned char U8;

char var = 0xf0;
char fn(char bar)
{
    char* ptr = (char*)0xf;
    var |= ptr;
    while (var > bar)
        var <<= 1;
    return 0;
}

void main() {
    fn(0x7f);
    assert(0);
}

Gets stuck in an endless loop with -O.
