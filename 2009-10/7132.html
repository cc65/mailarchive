    char foo;
} Bar;

void main() {
    Bar bar;
    Bar* const baz = &bar;
    baz->foo = 1;
}

This produces the compiler error "test.c(9): Error: Assignment to const"

Shouldn't be an error, should it? baz is const, bar isn't.
