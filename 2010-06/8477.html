This little Makefile-hack seems to work well enough for my purposes:

%.o: %.s
	sed -r 's/\s+:\s+/\n\t/g' $< > $(<:.s=.s2)
	ca65 -U -g -l $(<:.s=.s2)
	rm $(<:.s=.s2)
