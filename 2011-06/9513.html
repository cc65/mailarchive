#include <stdio.h>
#include <string.h> 
int main()
{
    char f[20];
    printf ("Type in name of file to compress: ");
    fgets(f,17,stdin);
    putchar ('\n');
    strcat (f,",p");
    printf("\ntext = \"%s\"\n", f);
    return 0;
}

does work. Even if you probably want to strip newline from the input string first. Something like..

#include <stdio.h>
#include <string.h> 
int main()
{
    char f[20];
    char *newline;
    printf ("Type in name of file to compress: ");
    fgets(f,17,stdin);
    putchar ('\n');
    if(newline = strchr(f, '\n'))
        *newline = '\0';
    strcat (f,",p");
    printf("\ntext = \"%s\"\n", f);
    return 0;
}
