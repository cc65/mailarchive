void DoSierpinskiCarpet(uint16_t x0L, uint8_t y0L, uint16_t width, 
uint8_t hight)
{
     uint8_t k,b,h;

     if (width>2 && hight>2)
     {
         b = width/3;
         h = hight/3;
         tgi_bar (x0L+b, y0L+h,x0L+2*b, y0L+2*h);

         for (k=0; k<9; k++) if (k!=4)
             DoSierpinskiCarpet (x0L+(k/3)*b, y0L+(k%3)*h, b, h); //

     }
}
