Don't you just mean:

// A pointer to some open RAM
char *someMemory = 0x200;

typedef struct
{
   int a;
   int b;
   int c;
} Astruct;

typedef struct
{
   char a;
   char b;
   char c;
} Bstruct;

// Lay an Astruct and a Bstruct over the open RAM area.
Astruct* pA = someMemory;
BStruct* pB = someMemory;
