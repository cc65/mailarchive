So in my case I would end up with something like:

#define ROM0 0

if (rootdir(EEPROM)) {

   char *currentdrive = getcwd();
   chdir(rootdir(EEPROM));
   fd = open("0", O_RDONLY);
   read(fd, &highscoretable, sizeof(highscoretable));
   //close(fd);
   chdir(currentdrive);
}
